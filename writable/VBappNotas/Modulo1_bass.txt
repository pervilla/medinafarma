Public Sub MUESTRA_USUario()
FORMGEN.i_CODUSU.Clear
usu.Requery
usu.MoveFirst
FORMGEN.i_CODUSU.AddItem ""
Do Until usu.EOF
  FORMGEN.i_CODUSU.AddItem usu!usu_key ' & "      " & par!PAR_CODCIA
  usu.MoveNext
Loop
End Sub

Dim NUMCAMPO As Integer
Dim OJO As String * 1
Static P As Boolean
Dim var As String
Dim sw_cuenta As Integer

'FORMGEN.LEIDO2.SetFocus
On Error GoTo CHECKERROR
If Left(TEXTO, 1) = "%" Then
  TEXTO = Mid(TEXTO, 2, Len(TEXTO))
  archi = "SELECT CLI_CODCLIE, CLI_CODCIA, CLI_CP, CLI_NOMBRE_EMPRESA,CLI_CASA_DIREC,CLI_ZONA_NEW, CLI_CASA_NUM,TAB_NOMLARGO FROM CLIENTES,TABLAS WHERE  CLI_CP = '" & wcp & "' AND CLI_CODCIA = '" & LK_CODCIA & "' AND (TAB_CODCIA = '00') AND (CLI_ZONA_NEW = TAB_NUMTAB) AND TAB_TIPREG = 35 AND CLI_NOMBRE_EMPRESA LIKE '%" & TEXTO & "%' ORDER BY CLI_NOMBRE_EMPRESA"
  GoTo LISTO_SELE
End If

If Not IsMissing(WBUS) Then
  If WBUS = "D" Then
  archi = "SELECT CLI_CODCLIE, CLI_CODCIA, CLI_CP, CLI_NOMBRE,CLI_CASA_DIREC,CLI_ZONA_NEW, CLI_CASA_NUM,TAB_NOMLARGO FROM CLIENTES,TABLAS WHERE  CLI_CP = '" & wcp & "' AND CLI_CODCIA = '" & LK_CODCIA & "' AND (TAB_CODCIA = '00') AND (CLI_ZONA_NEW = TAB_NUMTAB) AND TAB_TIPREG = 35 AND CLI_CASA_DIREC LIKE '%" & TEXTO & "%' ORDER BY CLI_NOMBRE"
  End If
Else
  archi = "SELECT CLI_CODCLIE, CLI_CODCIA, CLI_CP, CLI_NOMBRE,CLI_CASA_DIREC,CLI_ZONA_NEW, CLI_CASA_NUM,TAB_NOMLARGO FROM CLIENTES,TABLAS WHERE  CLI_CP = '" & wcp & "' AND CLI_CODCIA = '" & LK_CODCIA & "' AND (TAB_CODCIA = '00') AND (CLI_ZONA_NEW = TAB_NUMTAB) AND TAB_TIPREG = 35 AND CLI_NOMBRE LIKE '%" & TEXTO & "%' ORDER BY CLI_NOMBRE"
End If
LISTO_SELE:
Set PSX = CN.CreateQuery("", archi)
Set X = PSX.OpenResultset(rdOpenForwardOnly)
X.Requery
fila = 0
WLV1.ColumnHeaders.Add 1, , "Descripci贸n", 3800
WLV1.ColumnHeaders.Add 2, , "Cod.", 600
WLV1.ColumnHeaders.Add 3, , "Direcci贸n", 3200

WLV1.Width = 11000
WLV1.Left = 300

NUMCAMPO = 0
var = "*" & TEXTO & "*"
sw_cuenta = 0
OJO = "S"
Do Until X.EOF Or sw_cuenta = 1000
   Set itmX = WLV1.ListItems.Add(, , Trim(CStr(X.rdoColumns(3))))
   itmX.SubItems(1) = Trim(CStr(X.rdoColumns(0)))
   itmX.SubItems(2) = Trim(CStr(X.rdoColumns(4))) + " # " + Trim(CStr(X.rdoColumns(6)))
   itmX.SubItems(3) = Trim(CStr(X!TAB_NOMLARGO))
   NUMCAMPO = 1
   sw_cuenta = sw_cuenta + 1
   itmX.Tag = sw_cuenta
   X.MoveNext
Loop
WLV1.ToolTipText = "Encontrados : " & itmX.Tag & "/" & sw_cuenta & " Muestra un Maximo de: " & 1000

alta_vista_nombre = NUMCAMPO

GoTo fin
CHECKERROR:

End Function

Public Function ENTERO(TEXTO As String) As Boolean
Dim LARGO As Integer
Dim I, X As Integer
Dim DIG As Integer
LARGO = Len(TEXTO)
I = LARGO
ENTERO = True
Do Until I = 0
   DIG = Asc(Mid(TEXTO, I, 1))
   If (DIG > 47 And DIG < 58) Then
       X = 0
   Else
       ENTERO = False
       Exit Do
   End If
   I = I - 1
  
   Loop

End Function
